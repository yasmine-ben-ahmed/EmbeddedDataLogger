28#define TRC_STRCAT(dst, dst_size, pDstLength, src) \
29 { \
30 TraceUnsignedBaseType_t uxTRC_STRCAT_INDEX = 0; \
31 while (*(pDstLength) < (dst_size)) \
32 { \
33 dst[*(pDstLength)] = src[uxTRC_STRCAT_INDEX]; \

36 (*(pDstLength))++; \
37 uxTRC_STRCAT_INDEX++; \
38 } \
39 }
